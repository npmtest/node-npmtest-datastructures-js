                        if (currentNode.getRight() === null) {
                            currentNode.setRight(node);
                        }
                        else {
                            insertNode(currentNode.getRight(), node);
                        }

                        if (currentNode.getLeft() === null) {
                            currentNode.setLeft(node);
                        }
                        else {
                            insertNode(currentNode.getLeft(), node);
                        }

                        return null;

                        self.nodesCount--;
                        return node.getLeft();
